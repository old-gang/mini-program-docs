# 路由传参

  • A 页面
Page({
  // 事件处理函数
  handler() {
    wx.navigateTo({
      url: '/pages/logs/logs?name=zs&age=18',
    })
  },
})
  • B 页面(logs 页面)
Page({
  // 页面加载成功
  onLoad(options) {
    console.log(options) // {name:'zs',age:18}
  },
})

# 事件通道

基础库 2.7.3 开始支持。 如果一个页面由另一个页面通过 wx.navigateTo 打开，这两个页面间将建立一条数据通道：被打开的页面可以通过 this.getOpenerEventChannel() 方法来获得一个 EventChannel 对象； wx.navigateTo 的 success 回调中也包含一个 EventChannel 对象。 这两个 EventChannel 对象间可以使用 emit 和 on 方法相互发送、监听事件。

  • A 页面
Page({
  handler(e) {
    wx.navigateTo({
      url: '/pages/logs/logs?name=zs&age=18',
      events: {
        // 为指定事件添加一个监听器，获取被打开页面传送到当前页面的数据
        acceptDataFromOpenedPage: function (data) {
          console.log('index', data)
        },
        someEvent: function (data) {
          console.log('index', data)
        },
      },
      success: function (res) {
        // 通过eventChannel向被打开页面传送数据
        console.log('index', '切换成功')
        res.eventChannel.emit('acceptDataFromOpenerPage', {
          data: 'index-test',
        })
      },
    })
  },
})
  • B 页面(logs 页面)
Page({
  onLoad: function (option) {
    console.log(option) //{name: "zs", age: "18"}
    const eventChannel = this.getOpenerEventChannel()
    // 触发index页面中通过eventChannel绑定的两个事件,传递参数
    eventChannel.emit('acceptDataFromOpenedPage', { data: 'logs-test' })
    eventChannel.emit('someEvent', { data: 'logs-test' })
    // 监听acceptDataFromOpenerPage事件，获取上一页面通过eventChannel传送到当前页面的数据
    eventChannel.on('acceptDataFromOpenerPage', function (data) {
      console.log('logs', data)
    })
  },
})
